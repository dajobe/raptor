# Expected failures are documented with the matching Autotools lists in
# Makefile.am. They are all unsupported RDF 1.1 TriG grammar features.
IF(RAPTOR_PARSER_TRIG AND RAPTOR_SERIALIZER_NQUADS)
  FOREACH(TEST_FILE ${TEST_GOOD_FILES})
    GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
    SET(TEST_PREFIX trig-2013.good.${TEST_NAME})
    RAPPER_PARSE_TEST(${TEST_PREFIX}
      "${RAPPER} -q -i trig -o nquads file:${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
    )
    LIST(FIND EXPECTED_GOOD_FAILURES ${TEST_FILE} EXPECTED_INDEX)
    IF(NOT EXPECTED_INDEX EQUAL -1)
      SET_TESTS_PROPERTIES(${TEST_PREFIX} PROPERTIES WILL_FAIL TRUE)
    ENDIF()
  ENDFOREACH()

  FOREACH(TEST_FILE ${TEST_BAD_FILES})
    GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
    RAPPER_EXPECT_FAIL_TEST(trig-2013.bad-syntax.${TEST_NAME}
      "${RAPPER} -q -i trig -o nquads file:${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
    )
  ENDFOREACH()

  FOREACH(TEST_CASE ${TEST_EVAL_CASES})
    STRING(REPLACE "|" ";" TEST_FIELDS ${TEST_CASE})
    LIST(GET TEST_FIELDS 0 TEST_FILE)
    LIST(GET TEST_FIELDS 1 EXPECTED_FILE)
    GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
    SET(TEST_PREFIX trig-2013.eval.${TEST_NAME})
    LIST(FIND EXPECTED_EVAL_FAILURES ${TEST_FILE} EXPECTED_INDEX)
    LIST(FIND RDFDIFF_EVAL_FILES ${TEST_FILE} RDFDIFF_INDEX)
    IF(NOT EXPECTED_INDEX EQUAL -1)
      RAPPER_RDFDIFF_TEST(${TEST_PREFIX}
        "${RAPPER} -q -i trig -o nquads ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
        ${TEST_PREFIX}.res
        "${RDFDIFF} -b -f nquads -u http://www.w3.org/2013/TriGTests/${TEST_FILE} -t nquads ${CMAKE_CURRENT_SOURCE_DIR}/${EXPECTED_FILE} ${TEST_PREFIX}.res"
        EXPECTED_FAILURE
      )
    ELSEIF(NOT RDFDIFF_INDEX EQUAL -1)
      RAPPER_RDFDIFF_TEST(${TEST_PREFIX}
        "${RAPPER} -q -i trig -o nquads ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
        ${TEST_PREFIX}.res
        "${RDFDIFF} -b -f nquads -u http://www.w3.org/2013/TriGTests/${TEST_FILE} -t nquads ${CMAKE_CURRENT_SOURCE_DIR}/${EXPECTED_FILE} ${TEST_PREFIX}.res"
      )
    ELSE()
      RAPPER_COMPARE_TEST(${TEST_PREFIX}
        "${RAPPER} -q -i nquads -o nquads ${CMAKE_CURRENT_SOURCE_DIR}/${EXPECTED_FILE}"
        "${RAPPER} -q -i trig -o nquads ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
      )
    ENDIF()
  ENDFOREACH()

  FOREACH(TEST_FILE ${TEST_BAD_EVAL_FILES})
    GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
    RAPPER_EXPECT_FAIL_TEST(trig-2013.bad-eval.${TEST_NAME}
      "${RAPPER} -q -i trig -o nquads file:${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} http://www.w3.org/2013/TriGTests/${TEST_FILE}"
    )
  ENDFOREACH()
ENDIF()
