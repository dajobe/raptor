SET(BASE_URI http://www.w3.org/2013/TurtleTests/)

IF(RAPTOR_PARSER_TURTLE AND RAPTOR_SERIALIZER_NTRIPLES)
	FOREACH(TEST_FILE ${TEST_GOOD_FILES})
		GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
		SET(TEST_PREFIX turtle-2013.good.${TEST_NAME})
		RAPPER_PARSE_TEST(${TEST_PREFIX}
			"${RAPPER} -q -i turtle -o ntriples file:${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} ${BASE_URI}${TEST_FILE}"
		)
	ENDFOREACH(TEST_FILE)

	SET(ALL_BAD_FILES ${TEST_BAD_FILES} ${TEST_BAD_EVAL_FILES})
	FOREACH(TEST_FILE ${ALL_BAD_FILES})
		GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
		SET(TEST_PREFIX turtle-2013.bad.${TEST_NAME})
		RAPPER_EXPECT_FAIL_TEST(${TEST_PREFIX}
			"${RAPPER} -q -i turtle -o ntriples file:${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} ${BASE_URI}${TEST_FILE}"
		)
	ENDFOREACH(TEST_FILE)

	FOREACH(TEST_FILE ${TEST_EVAL_FILES})
		GET_FILENAME_COMPONENT(TEST_NAME ${TEST_FILE} NAME_WE)
		SET(TEST_PREFIX turtle-2013.eval.${TEST_NAME})
		RAPPER_COMPARE_TEST(${TEST_PREFIX}
			"${RAPPER} -q -i turtle -o ntriples ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_NAME}.nt"
			"${RAPPER} -q -i turtle -o ntriples ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_FILE} ${BASE_URI}${TEST_FILE}"
		)
	ENDFOREACH(TEST_FILE)
ENDIF(RAPTOR_PARSER_TURTLE AND RAPTOR_SERIALIZER_NTRIPLES)
